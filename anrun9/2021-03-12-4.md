def algorithm(self, N, isles):
  rr = [0, 1, 1, 1, 0, -1, -1, -1]
  cc = [1, 1, 0, -1, -1, -1, 0, 1]
  res = 0
  queue = deque()
  for i in range(N):
    for j in range(N):
      if isles[i][j] == 1:
        queue.append((i, j))
        isles[i][j] = 0

        while queue:
          r, c = queue.popleft()
          isles[r][c] = 0
          for k in range(8):
            nr = r + rr[k]
            nc = c + cc[k]
            if 0 <= nr < N and 0 <= nc < N:
              if isles[nr][nc] == 1:
                queue.append((nr, nc))
                
        res += 1
        
  return res
